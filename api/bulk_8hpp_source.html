1// ============================================================================
2// nanom/bulk.hpp — data-parallel bulk decode into columnar (SoA) storage.
3//
4// The model is the one a GPU wants and nanotins uses: a grid of independent
5// tasks, one per packet, each decoding its packet and scattering the result
6// into pre-sized SoA columns at its own row index — *disjoint* writes, so no
7// locks, no atomics, no push_back, no per-task allocation. On CPU the grid is a
8// thread pool (`par_exec`); on a device it is a kernel launch. The per-packet
9// work (`decode`) and the field scatter (`scatter_row`) are POD, allocation-free
10// and NANOM_HD-annotated, so the *same* code compiles for host and device.
11//
12// nm::bulk_table<PacketRow> tbl;
13// nm::bulk_decode(packets, tbl, kernel, nm::par_exec{}); // fills columns
14// auto ts = tbl.column<std::uint64_t>("ts_raw"); // contiguous span
15//
16// where `kernel` is bool(const nm::pkt_ref&, PacketRow&) — a device-safe
17// function the caller writes with nanom's overlay<>/get<> (see examples).
18//
19// GPU status: this header is written to be device-compilable (see docs/GPU.md);
20// the CPU `par_exec` path is what runs and is benchmarked here. Swapping in a
21// CUDA/HIP launcher for the executor is the only remaining step.
22// SPDX-License-Identifier: Apache-2.0
23// ============================================================================

53NANOM_HD inline void byte_copy(std::byte* dst, const void* src, std::size_t n) {
54 const std::byte* s = reinterpret_cast<const std::byte*>(src);
55 for (std::size_t k = 0; k < n; ++k) dst[k] = s[k];
56}

61template <class F>
62NANOM_HD void scatter_one(std::byte* const* col_base, const std::size_t* elem,
63 std::size_t i, std::size_t& c, const F& v) {
64 if constexpr (Described<F>) {
65 for_each_field<F>([&](auto f) { scatter_one(col_base, elem, i, c, v.*(decltype(f)::mem_ptr)); });
66 } else if constexpr (is_std_array_v<F>) {
67 using ED = typename wire<typename F::value_type>::decoded;
68 std::byte* dst = col_base[c] + i * elem[c];
69 std::size_t off = 0;
70 for (const auto& x : v) { ED d = ED(x); byte_copy(dst + off, &d, sizeof(ED)); off += sizeof(ED); }
71 ++c;
72 } else {
73 using D = typename wire<F>::decoded;
74 D d = D(v);
75 byte_copy(col_base[c] + i * elem[c], &d, sizeof(D));
76 ++c;
77 }
78}

80} // namespace detail

83template <Described Row>
84NANOM_HD void scatter_row(std::byte* const* col_base, const std::size_t* elem,
85 std::size_t i, const Row& r) {
86 std::size_t c = 0;
87 detail::for_each_field<Row>([&](auto f) {
88 detail::scatter_one(col_base, elem, i, c, r.*(decltype(f)::mem_ptr));
89 });
90}

98template <Described Row>
99class bulk_table {
100 static_assert(std::is_trivially_copyable_v<Row>,
101 "bulk_table Row must be trivially copyable (needed for device transfer)");
102 public:
103 using column_info = typename soa<Row>::column_info;
104
105 bulk_table() { columns_ = soa<Row>{}.columns(); }
106
107 const std::vector<column_info>& columns() const { return columns_; }
108 std::size_t capacity() const { return n_; }
109
111 void prepare(std::size_t n) {
112 n_ = n;
113 storage_.assign(columns_.size(), {});
114 base_.assign(columns_.size(), nullptr);
115 elem_.assign(columns_.size(), 0);
116 for (std::size_t c = 0; c < columns_.size(); ++c) {
117 storage_[c].assign(n * columns_[c].elem_bytes, std::byte{0});
118 base_[c] = storage_[c].data();
119 elem_[c] = columns_[c].elem_bytes;
120 }
121 valid_.assign(n, 0);
122 }
123
124 // --- the device-transferable sink (raw pointers a kernel writes through) ---
125 std::byte* const* col_base() const { return base_.data(); }
126 const std::size_t* col_elem() const { return elem_.data(); }
127 std::uint8_t* valid() { return valid_.data(); }
128 const std::uint8_t* valid() const { return valid_.data(); }
129
131 std::size_t rows() const {
132 std::size_t k = 0;
133 for (auto v : valid_) k += v;
134 return k;
135 }
137 template <class V>
138 std::span<const V> column(std::string_view name) const {
139 for (std::size_t c = 0; c < columns_.size(); ++c)
140 if (columns_[c].name == name)
141 return {reinterpret_cast<const V*>(storage_[c].data()), n_};
142 return {};
143 }
144 bytes column_bytes(std::size_t c) const { return {storage_[c].data(), storage_[c].size()}; }
145
146 private:
147 std::vector<column_info> columns_;
148 std::vector<std::vector<std::byte>> storage_;
149 std::vector<std::byte*> base_;
150 std::vector<std::size_t> elem_;
151 std::vector<std::uint8_t> valid_;
152 std::size_t n_ = 0;
153};

156// executors — the "grid". Each runs a functor over [0, n) once per index.

160struct seq_exec {
161 template <class F>
162 void run(std::size_t n, F&& f) const {
163 for (std::size_t i = 0; i < n; ++i) f(i);
164 }
165};

169struct par_exec {
170 unsigned threads = 0;
171 template <class F>
172 void run(std::size_t n, F&& f) const {
173 unsigned t = threads ? threads : std::max(1u, std::thread::hardware_concurrency());
174 if (t <= 1 || n < 4096) { for (std::size_t i = 0; i < n; ++i) f(i); return; }
175 const std::size_t chunk = (n + t - 1) / t;
176 std::vector<std::thread> pool;
177 pool.reserve(t);
178 for (unsigned k = 0; k < t; ++k) {
179 const std::size_t lo = std::size_t(k) * chunk, hi = std::min(n, lo + chunk);
180 if (lo >= hi) break;
181 pool.emplace_back([&f, lo, hi] { for (std::size_t i = lo; i < hi; ++i) f(i); });
182 }
183 for (auto& th : pool) th.join();
184 }
185};

187// ---------------------------------------------------------------------------
188// the driver: one row per packet, filled in parallel with disjoint writes.
189// `kernel(pkt, row) -> bool` decodes packet -> row; false = leave row default
190// and mark invalid. Returns the number of valid rows produced.
191// ---------------------------------------------------------------------------
192template <Described Row, class Kernel, class Exec = par_exec>
193std::size_t bulk_decode(std::span<const pkt_ref> pkts, bulk_table<Row>& tbl,
194 Kernel kernel, Exec exec = {}) {
195 for (const pkt_ref& pk : pkts)
196 if (!pkt_ref_valid(pk)) return 0;
197 tbl.prepare(pkts.size());
198 std::byte* const* base = tbl.col_base();
199 const std::size_t* elem = tbl.col_elem();
200 std::uint8_t* valid = tbl.valid();
201 const pkt_ref* p = pkts.data();
202 exec.run(pkts.size(), [=](std::size_t i) {
203 Row r{};
204 if (kernel(p[i], r)) { // pure, device-safe, no allocation
205 scatter_row<Row>(base, elem, i, r);
206 valid[i] = 1;
207 }
208 });
209 return tbl.rows();
210}
