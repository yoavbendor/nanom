177enum class errk : std::uint8_t { err, fail, incomplete };
178
185inline constexpr std::size_t max_context = 4;
188inline constexpr std::uint32_t max_incomplete_needed = 64 * 1024;
189
190struct error {
191 struct frame { const char* label; std::uint32_t offset; };
192
193 errk kind = errk::err;
194 std::uint8_t nctx = 0;
195 std::uint32_t offset = 0;
196 const char* expected = "";
197 std::uint32_t needed = 0;
198 std::array<frame, max_context> ctx{};
199
200 constexpr void push_context(const char* label, std::size_t off) {
201 if (nctx < ctx.size()) ctx[nctx++] = {label, std::uint32_t(off)};
202 }
203
206 std::string render(input whole) const {
207 std::string out;
208 out += "parse ";
209 out += kind == errk::incomplete ? "incomplete" : kind == errk::fail ? "failure" : "error";
210 out += " at offset " + std::to_string(offset);
211 if (needed)
212 out += " (need " + std::to_string(needed) + " more byte(s))";
213 out += ": expected ";
214 out += expected;
215 for (std::uint8_t i = 0; i < nctx; ++i) {
216 out += "\n in ";
217 out += ctx[i].label;
218 out += " (starting at offset " + std::to_string(ctx[i].offset) + ")";
219 }
220 // hex window: up to 8 bytes before and after the failure point
221 const std::size_t total = whole.base ? std::size_t(whole.last - whole.base) : 0;
222 const std::size_t off = std::min<std::size_t>(offset, total); // clamp bogus offsets
223 if (offset > total)
224 out += " (offset beyond input, hex window clamped)";
225 if (total > 0) {
226 const std::size_t lo = off >= 8 ? off - 8 : 0;
227 const std::size_t hi = std::min(off + 8, total);
228 static constexpr char hexd[] = "0123456789abcdef";
229 std::string line = " ", caret = " ";
230 for (std::size_t i = lo; i < hi; ++i) {
231 const auto b = std::uint8_t(whole.base[i]);
232 line += hexd[b >> 4]; line += hexd[b & 15]; line += ' ';
233 caret += (i == off) ? "^^ " : " ";
234 }
235 if (off == total) caret += "^^ (end of input)";
236 out += "\n" + line + "\n" + caret;
237 }
238 return out;
239 }
240};

303inline constexpr std::size_t tag_sbo_cap = 15;
304constexpr char lower(char c) { return (c >= 'A' && c <= 'Z') ? char(c + 32) : c; }
305
308struct owned_pattern {
309 std::array<char, 16> small{};
310 std::string large;
311 std::size_t len = 0;
312
313 explicit owned_pattern(std::string_view p) : len(p.size()) {
314 if (len <= tag_sbo_cap) {
315 for (std::size_t i = 0; i < len; ++i) small[static_cast<std::size_t>(i)] = p[i];
316 } else {
317 large.assign(p);
318 }
319 }
320
321 const char* data() const noexcept {
322 return len <= tag_sbo_cap ? small.data() : large.data();
323 }
324};
325
326inline auto make_tag_parser(owned_pattern pat) {
327 return [pat = std::move(pat)](input in) -> result<bytes> {
328 const std::size_t n = pat.len;
329 if (in.size() < n) return make_incomplete(in, n - in.size());
330 if (std::memcmp(in.first, pat.data(), n) != 0) return make_err(in, "tag");
331 return done{in.take_span(n), in.advance(n)};
332 };
333}
334
335inline auto make_tag_no_case_parser(owned_pattern pat) {
336 return [pat = std::move(pat)](input in) -> result<bytes> {
337 const std::size_t n = pat.len;
338 if (in.size() < n) return make_incomplete(in, n - in.size());
339 for (std::size_t i = 0; i < n; ++i)
340 if (lower(char(in[i])) != lower(pat.data()[i])) return make_err(in, "tag_no_case");
341 return done{in.take_span(n), in.advance(n)};
342 };
343}
344} // namespace detail

375template <class Pred>
376constexpr result<bytes> take_pred(input in, Pred&& pred, std::size_t min,
377 std::size_t max, const char* what) {
378 std::size_t i = 0;
379 const std::size_t cap = std::min(max, in.size());
380 while (i < cap && pred(std::uint8_t(in[i]))) ++i;
381 if (i < min) {
382 if (i == in.size()) return make_incomplete(in, min - i);
383 return make_err(in.advance(i), what);
384 }
385 return done{in.take_span(i), in.advance(i)};
386}
387constexpr std::size_t nmax = std::numeric_limits<std::size_t>::max();
388} // namespace detail

427constexpr auto take_until(std::string_view needle) {
428 return [needle](input in) -> result<bytes> {
429 auto hay = in.span();
430 auto ned = bytes(reinterpret_cast<const std::byte*>(needle.data()), needle.size());
431 auto it = std::search(hay.begin(), hay.end(), ned.begin(), ned.end());
432 if (it == hay.end()) return make_incomplete(in, 1);
433 const std::size_t n = std::size_t(it - hay.begin());
434 return done{in.take_span(n), in.advance(n)};
435 };

574template <class Tup, Parser P, Parser... Ps>
575constexpr auto seq_step(Tup&& acc, input in, const P& p, const Ps&... ps)
576 -> result<decltype(std::tuple_cat(std::move(acc), std::declval<std::tuple<parsed_t<P>, parsed_t<Ps>...>>()))> {

580 if constexpr (sizeof...(Ps) == 0)

584}

664template <class V, Parser P, Parser... Ps>
665constexpr result<V> alt_step(input in, const error* best, const P& p, const Ps&... ps) {
666 auto r = p(in);

668 if (r.error().kind != errk::err) return unexp(r.error()); // failure/incomplete: stop

670 if constexpr (sizeof...(Ps) == 0) {

672 if (out.expected[0] == '\0') out.expected = "one of the alternatives";
673 return unexp(out);
674 } else {
675 return alt_step<V>(in, &e, ps...);
676 }

693template <Parser... Ps>
694 requires(sizeof...(Ps) >= 1)
695constexpr auto permutation(Ps... ps) {
696 return [ps...](input in) -> result<std::tuple<parsed_t<Ps>...>> {
697 constexpr std::size_t N = sizeof...(Ps);
698 std::tuple<std::optional<parsed_t<Ps>>...> slots;
699 auto parsers = std::tie(ps...);
700 input cur = in;
701 error deepest{};
702 for (std::size_t round = 0; round < N; ++round) {
703 bool progressed = false;
704 expected<input, error> abort_err = cur; // reused as failure carrier
705 auto try_one = [&]<std::size_t I>(std::integral_constant<std::size_t, I>) {
706 if (progressed || std::get<I>(slots)) return true;
707 auto r = std::get<I>(parsers)(cur);
708 if (r) {
709 std::get<I>(slots).emplace(std::move(r->value));
710 cur = r->rest;
711 progressed = true;
712 } else if (r.error().kind != errk::err) {
713 abort_err = unexp(r.error());
714 return false;
715 } else {
716 deepest = detail::better(deepest, r.error());
717 }
718 return true;
719 };
720 bool okrun = [&]<std::size_t... Is>(std::index_sequence<Is...>) {
721 return (try_one(std::integral_constant<std::size_t, Is>{}) && ...);
722 }(std::make_index_sequence<N>{});
723 if (!okrun) return unexp(abort_err.error());
724 if (!progressed) {
725 error e = deepest;
726 if (e.expected[0] == '\0') { e.offset = cur.offset(); e.expected = "permutation alternative"; }
727 return unexp(e);
728 }
729 }
730 return done{[&]<std::size_t... Is>(std::index_sequence<Is...>) {
731 return std::tuple<parsed_t<Ps>...>{std::move(*std::get<Is>(slots))...};
732 }(std::make_index_sequence<N>{}), cur};
733 };

1201 std::conditional_t<Bytes <= 1, std::uint8_t,
1202 std::conditional_t<Bytes <= 2, std::uint16_t,
1203 std::conditional_t<Bytes <= 4, std::uint32_t, std::uint64_t>>>;

1207template <std::size_t Bytes>
1208constexpr result<uint_for_bytes<Bytes>> read_uint(input in, std::endian order) {
1209 static_assert(Bytes >= 1 && Bytes <= 8);
1210 if (in.size() < Bytes) return make_incomplete(in, Bytes - in.size());
1211 using U = uint_for_bytes<Bytes>;
1212 U v = 0;
1213 if (order == std::endian::big)
1214 for (std::size_t i = 0; i < Bytes; ++i) v = U(v << 8) | in[i];
1215 else
1216 for (std::size_t i = 0; i < Bytes; ++i) v |= U(U(in[i]) << (8 * i));
1217 return done{v, in.advance(Bytes)};
1218}

1338// integer overloads but not the floating-point ones, so `std::from_chars(b, e, double&)` there

1345// 10^0 .. 10^22 are each exactly representable as double.
1346inline constexpr double pow10_exact[] = {
1347 1e0, 1e1, 1e2, 1e3, 1e4, 1e5, 1e6, 1e7, 1e8, 1e9, 1e10, 1e11,
1348 1e12, 1e13, 1e14, 1e15, 1e16, 1e17, 1e18, 1e19, 1e20, 1e21, 1e22};
1349
1350// m * 10^e. On the classic fast path (|e| <= 22 and m exact as a double, i.e. <= 15 significant
1351// digits) this is a single correctly-rounded multiply/divide; with more digits or larger magnitudes
1352// it scales in 10^22 chunks and can differ from a correctly-rounded conversion by a few ulp (e.g. at
1353// full-precision DBL_MAX). Only the fallback stdlib (libc++) takes this path; libstdc++/MSVC keep
1354// std::from_chars. A full Eisel-Lemire/big-integer path is deliberately not carried here.
1355constexpr double scale10(double m, int e) {
1356 if (m == 0.0) return 0.0;
1357 if (e >= 0) {
1358 while (e > 22) { m *= 1e22; e -= 22; }
1359 return m * pow10_exact[e];
1360 }
1361 int k = -e;
1362 while (k > 22) { m /= 1e22; k -= 22; }
1363 return m / pow10_exact[k];
1364}

1366// Portable std::from_chars(double) for stdlibs lacking it. Matches from_chars' general-format
1367// grammar: optional leading '-', at least one decimal digit, optional '.', optional [eE][+-]?digits;
1368// no leading '+' or whitespace. ptr points one past the last consumed character; result_out_of_range
1369// on overflow/underflow to inf/0.
1370constexpr std::from_chars_result from_chars_double(const char* first, const char* last, double& out) {
1371 const char* p = first;
1372 bool neg = false;
1373 if (p != last && *p == '-') { neg = true; ++p; }
1374
1375 std::uint64_t mant = 0; // significand; digits capped so it can never overflow
1376 int digits = 0;
1377 int exp10 = 0;
1378 bool any = false;
1379 auto eat = [&](bool fractional) {
1380 for (; p != last && *p >= '0' && *p <= '9'; ++p) {
1381 any = true;
1382 if (digits < 18) { mant = mant * 10 + std::uint64_t(*p - '0'); ++digits; if (fractional) --exp10; }
1383 else if (!fractional) ++exp10; // extra integer digits scale up; extra fraction digits drop
1384 }
1385 };
1386 eat(false);
1387 if (p != last && *p == '.') { ++p; eat(true); }
1388 if (!any) return {first, std::errc::invalid_argument}; // from_chars needs at least one digit
1389
1390 if (p != last && (*p == 'e' || *p == 'E')) {
1391 const char* ep = p + 1;
1392 bool eneg = false;
1393 if (ep != last && (*ep == '+' || *ep == '-')) { eneg = (*ep == '-'); ++ep; }
1394 if (ep != last && *ep >= '0' && *ep <= '9') {
1395 int e = 0;
1396 for (; ep != last && *ep >= '0' && *ep <= '9'; ++ep) e = e < 100000 ? e * 10 + (*ep - '0') : e;
1397 exp10 += eneg ? -e : e;
1398 p = ep;
1399 }
1400 // else: a bare 'e' with no exponent digits is not part of the number; leave p pointing at it.
1401 }
1402
1403 double v = scale10(double(mant), exp10);
1404 const bool overflow = (v != 0.0 && v * 2.0 == v); // v == +inf
1405 const bool underflow = (v == 0.0 && mant != 0);
1406 if (overflow || underflow) return {p, std::errc::result_out_of_range};
1407 out = neg ? -v : v;
1408 return {p, std::errc{}};
1409}

1414// function even a discarded `if constexpr` branch is still type-checked, which re-triggers the

1482NANOM_HD constexpr bresult<std::uint64_t> read_bits(bit_input bi, std::size_t n, bit_order ord) {
1483 if (bi.bits_left() < n) return make_incomplete(bi.in, (n - bi.bits_left() + 7) / 8);
1484 std::uint64_t out = 0;
1485 input cur = bi.in;
1486 std::size_t bit = bi.bit;
1487 for (std::size_t got = 0; got < n;) {
1488 const std::uint8_t byte = cur[0];
1489 const std::size_t avail = 8 - bit;
1490 const std::size_t use = std::min(avail, n - got);
1491 std::uint64_t piece;
1492 if (ord == bit_order::msb0) {
1493 // take `use` bits starting at position (7 - bit) downward
1494 piece = (byte >> (avail - use)) & ((std::uint64_t(1) << use) - 1);
1495 out = (out << use) | piece;
1496 } else {
1497 // take `use` bits starting at position `bit` upward
1498 piece = (byte >> bit) & ((std::uint64_t(1) << use) - 1);
1499 out |= piece << got;
1500 }
1501 got += use;
1502 bit += use;
1503 if (bit == 8) { bit = 0; cur = cur.advance(1); }
1504 }
1505 return bdone{out, bit_input{cur, bit}};
1506}
