21template <Described T>
22class soa {
23 public:
24 struct column_info {
25 std::string name;
26 dkind kind;
27 std::size_t elem_bytes;
28 std::string arrow;
29 };
30 struct chunk {
31 std::size_t rows = 0;
32 std::vector<std::vector<std::byte>> cols;
33 bytes col(std::size_t i) const { return {cols[i].data(), cols[i].size()}; }
35 template <class V>
36 std::span<const V> as(std::size_t i) const {
37 return {reinterpret_cast<const V*>(cols[i].data()), rows};
38 }
39 };
40
41 explicit soa(std::size_t chunk_rows = 65536) : chunk_rows_(chunk_rows) {
42 build_columns<T>("");
43 open_.cols.resize(columns_.size());
44 }
45
46 const std::vector<column_info>& columns() const { return columns_; }
47 std::size_t rows() const { return sealed_rows_ + open_.rows; }
48
49 void push(const T& v) {
50 std::size_t c = 0;
51 push_fields(v, c);
52 if (++open_.rows == chunk_rows_) seal();
53 }
54
56 template <class F>
57 void for_each_chunk(F&& f) const {
58 for (const auto& ch : sealed_) f(ch);
59 if (open_.rows) f(open_);
60 }
62 void seal() {
63 if (!open_.rows) return;
64 sealed_rows_ += open_.rows;
65 sealed_.push_back(std::move(open_));
66 open_ = {};
67 open_.cols.resize(columns_.size());
68 }
69
70 private:
71 template <class F>
72 void build_columns(std::string prefix, std::string_view name = "") {
73 if constexpr (Described<F>) {
74 build_nested<F>(name.empty() ? prefix : prefix + std::string(name) + ".");
75 } else {
76 schema_field s = detail::field_schema<F>(name);
77 column_info ci;
78 ci.name = prefix + std::string(name);
79 if (s.kind == dkind::fixed_bin || s.kind == dkind::list) {
80 // store fixed lists as flat fixed-size binary (elem×len bytes per row)
81 ci.kind = dkind::fixed_bin;
82 ci.elem_bytes = detail::wire<F>::bits / 8;
83 schema_field bin = s;
84 bin.kind = dkind::fixed_bin;
85 bin.size = std::uint32_t(ci.elem_bytes);
86 ci.arrow = arrow_format(bin);
87 } else {
88 ci.kind = s.kind;
89 ci.elem_bytes = sizeof(typename detail::wire<F>::decoded);
90 ci.arrow = arrow_format(s);
91 }
92 columns_.push_back(std::move(ci));
93 }
94 }
95 template <class F>
96 void build_nested(std::string prefix) {
97 detail::for_each_field<F>([&](auto f) {
98 build_columns<detail::member_t<decltype(f)::mem_ptr>>(prefix, decltype(f)::name.sv());
99 });
100 }
101
102 template <class F>
103 void push_one(const F& v, std::size_t& c) {
104 if constexpr (Described<F>) {
105 push_fields(v, c);
106 } else if constexpr (detail::is_std_array_v<F>) {
107 auto& col = open_.cols[c++];
108 for (const auto& e : v) {
109 using ED = typename detail::wire<typename F::value_type>::decoded;
110 ED d = ED(e);
111 const auto* b = reinterpret_cast<const std::byte*>(&d);
112 col.insert(col.end(), b, b + sizeof(ED));
113 }
114 } else {
115 using D = typename detail::wire<F>::decoded;
116 D d = D(v);
117 auto& col = open_.cols[c++];
118 const auto* b = reinterpret_cast<const std::byte*>(&d);
119 col.insert(col.end(), b, b + sizeof(D));
120 }
121 }
122 template <class F>
123 void push_fields(const F& v, std::size_t& c) {
124 detail::for_each_field<F>([&](auto f) { push_one(v.*(decltype(f)::mem_ptr), c); });
125 }
126
127 std::size_t chunk_rows_;
128 std::vector<column_info> columns_;
129 std::vector<chunk> sealed_;
130 chunk open_;
131 std::size_t sealed_rows_ = 0;
132};
