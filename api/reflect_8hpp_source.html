36template <class T, std::endian E>
37 requires(std::integral<T> || std::floating_point<T>)
38struct endian_scalar {
39 using value_type = T;
40 static constexpr std::endian order = E;
41 std::array<std::byte, sizeof(T)> raw{};
42
43 constexpr endian_scalar() = default;
44 constexpr endian_scalar(T v) { set(v); }
45
46 NANOM_HD constexpr T get() const {
47 using U = detail::uint_for_bytes<sizeof(T)>;
48 U u = 0;
49 if constexpr (E == std::endian::big)
50 for (std::size_t i = 0; i < sizeof(T); ++i) u = U(u << 8) | std::uint8_t(raw[i]);
51 else
52 for (std::size_t i = 0; i < sizeof(T); ++i) u |= U(U(std::uint8_t(raw[i])) << (8 * i));
53 if constexpr (std::floating_point<T>) return std::bit_cast<T>(u);
54 else return std::bit_cast<T>(U(u));
55 }
56 NANOM_HD constexpr void set(T v) {
57 using U = detail::uint_for_bytes<sizeof(T)>;
58 U u = std::bit_cast<U>(v);
59 if constexpr (E == std::endian::big)
60 for (std::size_t i = 0; i < sizeof(T); ++i)
61 raw[i] = std::byte((u >> (8 * (sizeof(T) - 1 - i))) & 0xff);
62 else
63 for (std::size_t i = 0; i < sizeof(T); ++i) raw[i] = std::byte((u >> (8 * i)) & 0xff);
64 }
65 NANOM_HD constexpr operator T() const { return get(); }
66};
67template <class T> using be = endian_scalar<T, std::endian::big>;
68template <class T> using le = endian_scalar<T, std::endian::little>;

194template <Described T>
195NANOM_HD constexpr auto field_bit_offsets() {
196 std::array<std::size_t, field_count_v<T>> off{};
197 std::size_t cur = 0, i = 0;
198 for_each_field<T>([&](auto f) {
199 off[i++] = cur;
200 cur += wire<member_t<decltype(f)::mem_ptr>>::bits;
201 });
202 return off;
203}

207template <Described T>
208constexpr bool layout_ok() {
209 constexpr auto off = field_bit_offsets<T>();
210 bool ok = wire<T>::bits % 8 == 0;
211 std::size_t i = 0;
212 for_each_field<T>([&](auto f) {
213 ok = ok && (wire<member_t<decltype(f)::mem_ptr>>::is_bits || off[i] % 8 == 0);
214 ++i;
215 });
216 return ok;
217}

244template <class F>
245NANOM_HD constexpr typename wire<F>::decoded decode_field(const std::byte* p, std::size_t bitoff,
246 std::endian dflt) {
247 using D = typename wire<F>::decoded;
248 if constexpr (std::integral<F> || std::floating_point<F>) {
249 using U = uint_for_bytes<sizeof(F)>;
250 const std::byte* q = p + bitoff / 8;
251 U u = 0;
252 if (dflt == std::endian::big)
253 for (std::size_t i = 0; i < sizeof(F); ++i) u = U(u << 8) | std::uint8_t(q[i]);
254 else
255 for (std::size_t i = 0; i < sizeof(F); ++i) u |= U(U(std::uint8_t(q[i])) << (8 * i));
256 return std::bit_cast<F>(U(u));
257 } else if constexpr (requires { F::order; F::bits; }) { // ubits / ibits
258 using VT = typename F::value_type;
259 constexpr unsigned N = F::bits;
260 std::uint64_t raw;
261 if constexpr (F::order == bit_order::msb0 && (7u + N) <= 64u) {
262 // fast path: load the covering bytes as one big-endian word and
263 // shift+mask — O(bytes) instead of read_bits' O(bits) loop. This is the
264 // hot path for network headers (msb0) and is what makes overlay<>()
265 // competitive with a hand-tuned overlay parser.
266 const std::byte* q = p + bitoff / 8;
267 const unsigned startbit = unsigned(bitoff % 8);
268 const unsigned nbytes = (startbit + N + 7) / 8;
269 std::uint64_t w = 0;
270 for (unsigned i = 0; i < nbytes; ++i) w = (w << 8) | std::uint8_t(q[i]);
271 const unsigned shift = nbytes * 8 - startbit - N;
272 raw = (w >> shift) & (N >= 64 ? ~std::uint64_t(0) : ((std::uint64_t(1) << N) - 1));
273 } else {
274 input fake{p + bitoff / 8, p + (bitoff + N + 7) / 8, p};
275 raw = read_bits(bit_input{fake, bitoff % 8}, N, F::order)->value;
276 }
277 if constexpr (std::is_signed_v<VT>) {
278 using UV = std::make_unsigned_t<VT>;
279 UV u = UV(raw);
280 const UV sign = UV(1) << (N - 1);
281 if (u & sign) u |= ~((sign << 1) - 1); // sign-extend
282 return std::bit_cast<VT>(u);
283 } else {
284 return VT(raw);
285 }
286 } else if constexpr (requires { F::order; typename F::value_type; }) { // be/le
287 // assemble the host value directly from the wire bytes in a single pass —
288 // for widths 2/4/8 the compiler lowers this to one load + bswap, matching
289 // a hand-tuned overlay parser (the old path copied to a temp, then that
290 // temp re-assembled: two loops on the hottest reads — ethertype, ports…).
291 using T = typename F::value_type;
292 using U = uint_for_bytes<sizeof(T)>;
293 const std::byte* q = p + bitoff / 8;
294 U u = 0;
295 if constexpr (F::order == std::endian::big)
296 for (std::size_t i = 0; i < sizeof(T); ++i) u = U(U(u << 8) | std::uint8_t(q[i]));
297 else
298 for (std::size_t i = 0; i < sizeof(T); ++i) u |= U(U(std::uint8_t(q[i])) << (8 * i));
299 return std::bit_cast<T>(u);
300 } else if constexpr (is_std_array_v<F>) {
301 using E = typename F::value_type;
302 D out{};
303 constexpr std::size_t eb = wire<E>::bits;
304 for (std::size_t i = 0; i < out.size(); ++i)
305 out[i] = decode_field<E>(p, bitoff + i * eb, dflt);
306 return out;
307 } else { // nested described struct
308 D out{};
309 constexpr auto offs = field_bit_offsets<F>();
310 std::size_t i = 0;
311 for_each_field<F>([&](auto f) {
312 out.*(decltype(f)::mem_ptr) =
313 assign_field<member_t<decltype(f)::mem_ptr>>(p, bitoff + offs[i++], dflt);
314 });
315 return out;
316 }
317}

321template <class F>
322NANOM_HD constexpr F assign_field(const std::byte* p, std::size_t bitoff, std::endian dflt) {
323 if constexpr (requires(F f) { f.raw; typename F::value_type; }) { // be/le: copy raw
324 F out;
325 for (std::size_t i = 0; i < out.raw.size(); ++i) out.raw[i] = p[bitoff / 8 + i];
326 return out;
327 } else if constexpr (requires { F::order; F::bits; }) { // ubits/ibits
328 return F{decode_field<F>(p, bitoff, dflt)};
329 } else if constexpr (is_std_array_v<F>) {
330 using E = typename F::value_type;
331 F out{};
332 constexpr std::size_t eb = wire<E>::bits;
333 for (std::size_t i = 0; i < out.size(); ++i)
334 out[i] = assign_field<E>(p, bitoff + i * eb, dflt);
335 return out;
336 } else if constexpr (std::integral<F> || std::floating_point<F>) {
337 return decode_field<F>(p, bitoff, dflt);
338 } else { // nested described struct
339 F out{};
340 constexpr auto offs = field_bit_offsets<F>();
341 std::size_t i = 0;
342 for_each_field<F>([&](auto f) {
343 out.*(decltype(f)::mem_ptr) =
344 assign_field<member_t<decltype(f)::mem_ptr>>(p, bitoff + offs[i++], dflt);
345 });
346 return out;
347 }

373template <Described T>
374constexpr auto strct(std::endian dflt = std::endian::native) {
375 static_assert(detail::layout_ok<T>(),
376 "nanom: bit fields must pack to byte boundaries and every "
377 "non-bit field must start byte-aligned");
378 return [dflt](input in) -> result<T> {
379 constexpr std::size_t need = wire_size_v<T>;
380 if (in.size() < need) return make_incomplete(in, need - in.size());
381 T out{};
382 constexpr auto offs = detail::field_bit_offsets<T>();
383 std::size_t i = 0;
384 detail::for_each_field<T>([&](auto f) {
385 out.*(decltype(f)::mem_ptr) =
386 detail::assign_field<detail::member_t<decltype(f)::mem_ptr>>(in.first, offs[i++], dflt);
387 });
388 return done{std::move(out), in.advance(need)};
389 };
390}

417template <Described T>
418struct view {
419 const std::byte* p = nullptr;
420 std::endian dflt = std::endian::native;
421
422 NANOM_HD constexpr bool valid() const noexcept { return p != nullptr; }
423
425 template <fixed_string Name>
426 NANOM_HD constexpr auto get() const {
427 detail::guard_view_pointer(p);
428 constexpr std::size_t I = detail::field_index<T, Name>();
429 static_assert(I != std::size_t(-1),
430 "nanom: no such field in this struct — check NANOM_DESCRIBE");
431 using F = detail::field_type_at<T, I>;
432 constexpr std::size_t off = detail::field_bit_offsets<T>()[I];
433 if constexpr (Described<F>) {
434 return view<F>{p + off / 8, dflt};
435 } else if constexpr (detail::is_byte_array_v<F>) {
436 // byte array (MAC / IPv4 / IPv6 address, name field…): the wire bytes ARE
437 // the value, so return a zero-copy span into the buffer instead of
438 // materializing a std::array. get<"src">()[0] is then a single byte load.
439 return std::span<const typename F::value_type, std::tuple_size_v<F>>(
440 reinterpret_cast<const typename F::value_type*>(p + off / 8), std::tuple_size_v<F>);
441 } else {
442 return detail::decode_field<F>(p, off, dflt);
443 }
444 }
446 NANOM_HD constexpr bytes raw() const {
447 detail::guard_view_pointer(p);
448 return {p, wire_size_v<T>};
449 }
451 NANOM_HD constexpr T to_struct() const {
452 detail::guard_view_pointer(p);
453 T out{};
454 constexpr auto offs = detail::field_bit_offsets<T>();
455 std::size_t i = 0;
456 detail::for_each_field<T>([&](auto f) {
457 out.*(decltype(f)::mem_ptr) =
458 detail::assign_field<detail::member_t<decltype(f)::mem_ptr>>(p, offs[i++], dflt);
459 });
460 return out;
461 }
462};

466constexpr auto overlay(std::endian dflt = std::endian::native) {
467 static_assert(detail::layout_ok<T>(),
