53// ---------------------------------------------------------------------------
54// Safety profile
55// ---------------------------------------------------------------------------
56// NANOM_STRICT is the "safe routes only" profile for safety- AND speed-obsessed
57// users. The idea is a *sweetened pill*: you accept compile-time restrictions,
58// and in return nanom compiles OUT the runtime safety machinery those
59// restrictions make unnecessary — so strict runs at the speed of the unchecked
60// `minimal` profile while the compiler (not a runtime branch) enforces the
61// lifetime contract.
62//
63// When NANOM_STRICT=1:
64// * runtime generation tracking and view guards are turned OFF by default
65// (lean input/bytes, no per-access generation branch) — see below.
66// * `[[clang::lifetimebound]]` is enforced on the zero-copy entry points, so
67// binding a view/span to a temporary buffer is a COMPILE error (the exact
68// dangling-view bug that generation tracking otherwise catches at runtime).
69// * the raw pointer+length `from()` escape hatch is removed (callers pass a
70// sized span/array/string_view instead — no unbounded provenance).
71// * the GPU/bulk data-parallel path (raw device-pointer scatter, outside the
72// bounds-checked combinator model) is unavailable: including <nanom/bulk.hpp>
73// under NANOM_STRICT is a hard error.
74// * combinator bounds checks are KEPT (defense against hostile input is not a
75// caller-discipline question) and the library stays continuously fuzzed.
76//
77// Everything is still individually overridable, e.g. -DNANOM_STRICT=1 with an
78// explicit -DNANOM_GENERATION=1 to keep runtime tracking on top of the strict
79// compile-time restrictions (paranoid belt-and-suspenders build).
80#ifndef NANOM_STRICT
81# define NANOM_STRICT 0
82#endif
83

106// NANOM_LIFETIMEBOUND — Clang's compile-time dangling-view diagnostic. Applied
107// to the zero-copy entry points (from(), as_str()) so a view/span bound to a
108// temporary buffer is diagnosed at compile time. Zero runtime cost; expands to
109// nothing on compilers without the attribute (e.g. GCC), which still build fine.
110// Under NANOM_STRICT the diagnostic is present whenever the toolchain supports
111// it; outside strict it is available but unobtrusive.
112#if defined(__has_cpp_attribute)
113# if __has_cpp_attribute(clang::lifetimebound)
114# define NANOM_LIFETIMEBOUND [[clang::lifetimebound]]
115# elif __has_cpp_attribute(msvc::lifetimebound)
116# define NANOM_LIFETIMEBOUND [[msvc::lifetimebound]]
117# endif
118#endif
119#ifndef NANOM_LIFETIMEBOUND
120# define NANOM_LIFETIMEBOUND
121#endif

123// NANOM_HD marks the functions on the zero-copy decode path as callable from a
124// GPU kernel. On a normal host build it expands to nothing (zero cost, zero API
125// change); under a CUDA/HIP compiler it becomes __host__ __device__ so the exact
126// same overlay/decode code runs on the device. The failure/formatting paths
127// (error::render, soa storage, the combinators that allocate) are deliberately
128// NOT annotated — they stay host-only. See docs/GPU.md and include/nanom/bulk.hpp.
129#ifndef NANOM_HD
130# if defined(__CUDACC__) || defined(__HIPCC__) || defined(__CUDA__)
131# define NANOM_HD __host__ __device__
132# else
133# define NANOM_HD
134# endif
135#endif
