42template <class D>
43constexpr dkind scalar_kind() {
44 if constexpr (std::is_same_v<D, float>) return dkind::f32;
45 else if constexpr (std::is_same_v<D, double>) return dkind::f64;
46 else if constexpr (std::is_signed_v<D>) {
47 if constexpr (sizeof(D) == 1) return dkind::i8;
48 else if constexpr (sizeof(D) == 2) return dkind::i16;
49 else if constexpr (sizeof(D) == 4) return dkind::i32;
50 else return dkind::i64;
51 } else {
52 if constexpr (sizeof(D) == 1) return dkind::u8;
53 else if constexpr (sizeof(D) == 2) return dkind::u16;
54 else if constexpr (sizeof(D) == 4) return dkind::u32;
55 else return dkind::u64;
56 }
57}

61template <class F>
62constexpr schema_field field_schema(std::string_view name) {
63 schema_field s{};
64 s.name = name;
65 s.bits = std::uint32_t(wire<F>::bits);
66 if constexpr (Described<F>) {
67 s.kind = dkind::record;
68 s.nested = &schema_holder<F>::node;
69 } else if constexpr (is_std_array_v<F>) {
70 using E = typename F::value_type;
71 using ED = typename wire<E>::decoded;
72 s.size = std::uint32_t(std::tuple_size_v<F>);
73 if constexpr (sizeof(ED) == 1 && std::is_integral_v<ED>) {
74 s.kind = dkind::fixed_bin; // byte arrays -> fixed binary
75 } else {
76 s.kind = dkind::list;
77 s.elem = scalar_kind<ED>();
78 }
79 } else {
80 s.kind = scalar_kind<typename wire<F>::decoded>();
81 }
82 return s;
83}

85template <Described T>
86constexpr auto make_schema_fields() {
87 std::array<schema_field, field_count_v<T>> out{};
88 std::size_t i = 0;
89 for_each_field<T>([&](auto f) {
90 out[i++] = field_schema<member_t<decltype(f)::mem_ptr>>(decltype(f)::name.sv());
91 });
92 return out;
93}

114inline std::string arrow_format(const schema_field& f) {
115 auto scalar = [](dkind k) -> const char* {
116 switch (k) {
117 case dkind::u8: return "C"; case dkind::u16: return "S";
118 case dkind::u32: return "I"; case dkind::u64: return "L";
119 case dkind::i8: return "c"; case dkind::i16: return "s";
120 case dkind::i32: return "i"; case dkind::i64: return "l";
121 case dkind::f32: return "f"; case dkind::f64: return "g";
122 default: return "z";
123 }
124 };
125 switch (f.kind) {
126 case dkind::fixed_bin: return "w:" + std::to_string(f.size);
127 case dkind::list: return "+w:" + std::to_string(f.size); // child: scalar(f.elem)
128 case dkind::record: return "+s";
129 default: return scalar(f.kind);
130 }
131}

135inline void avro_record_json(const schema_node& n, std::string& out) {
136 out += R"({"type":"record","name":")";
137 out += n.name;
138 out += R"(","fields":[)";
139 bool first = true;
140 for (const auto& f : n.fields) {
141 if (!first) out += ',';
142 first = false;
143 out += R"({"name":")";
144 out += f.name;
145 out += R"(","type":)";
146 avro_field_json(f, out);
147 out += '}';
148 }
149 out += "]}";
150}
151inline void avro_field_json(const schema_field& f, std::string& out) {
152 auto scalar = [](dkind k) -> const char* {
153 switch (k) {
154 case dkind::u8: case dkind::u16: case dkind::i8: case dkind::i16:
155 case dkind::i32: return R"("int")";
156 // u32 widens to long; u64 is emitted as long (Avro has no unsigned —
157 // values above 2^63 wrap; use fixed_bin if that matters)
158 case dkind::u32: case dkind::u64: case dkind::i64: return R"("long")";
159 case dkind::f32: return R"("float")";
160 case dkind::f64: return R"("double")";
161 default: return R"("bytes")";
162 }
163 };
164 switch (f.kind) {
165 case dkind::fixed_bin:
166 out += R"({"type":"fixed","name":")";
167 out += f.name;
168 out += R"(_fx","size":)" + std::to_string(f.size) + "}";
169 break;
170 case dkind::list:
171 out += R"({"type":"array","items":)";
172 out += scalar(f.elem);
173 out += '}';
174 break;
175 case dkind::record: avro_record_json(*f.nested, out); break;
176 default: out += scalar(f.kind);
177 }
178}

193template <class F>
194void json_value(const F& v, std::string& out) {
195 if constexpr (Described<F>) {
196 out += '{';
197 bool first = true;
198 for_each_field<F>([&](auto f) {
199 out += first ? "" : ",";
200 first = false;
201 out += '"';
202 out += decltype(f)::name.sv();
203 out += "\":";
204 json_value(v.*(decltype(f)::mem_ptr), out);
205 });
206 out += '}';
207 } else if constexpr (is_std_array_v<F>) {
208 using E = typename F::value_type;
209 using ED = typename wire<E>::decoded;
210 if constexpr (sizeof(ED) == 1 && std::is_integral_v<ED>) {
211 static constexpr char hexd[] = "0123456789abcdef"; // bytes -> hex string
212 out += '"';
213 for (auto e : v) {
214 auto b = std::uint8_t(typename wire<E>::decoded(e));
215 out += hexd[b >> 4]; out += hexd[b & 15];
216 }
217 out += '"';
218 } else {
219 out += '[';
220 bool first = true;
221 for (const auto& e : v) {
222 if (!first) out += ',';
223 first = false;
224 json_value(e, out);
225 }
226 out += ']';
227 }
228 } else if constexpr (std::floating_point<F>) {
229 char buf[32];
230 auto [p, ec] = std::to_chars(buf, buf + sizeof buf, double(v));
231 out.append(buf, p);
232 } else { // integral / be / le / ubits / ibits — all convert to a number
233 using D = typename wire<F>::decoded;
234 D d = D(v);
235 if constexpr (std::floating_point<D>) {
236 char buf[32];
237 auto [pe, ec] = std::to_chars(buf, buf + sizeof buf, double(d));
238 (void)ec;
239 out.append(buf, pe);
240 } else if constexpr (std::is_signed_v<D>) {
241 out += std::to_string(std::int64_t(d));
242 } else {
243 out += std::to_string(std::uint64_t(d));
244 }
245 }
246}

248template <class F>
249void csv_names(std::string prefix, std::string_view name, std::string& out, bool& first) {
250 if constexpr (Described<F>) {
251 std::string p2 = prefix + std::string(name) + ".";
252 for_each_field<F>([&](auto f) {
253 csv_names<member_t<decltype(f)::mem_ptr>>(p2, decltype(f)::name.sv(), out, first);
254 });
255 } else {
256 if (!first) out += ',';
257 first = false;
258 out += prefix;
259 out += name;
260 }
261}

263template <class F>
264void csv_value(const F& v, std::string& out, bool& first) {
265 if constexpr (Described<F>) {
266 for_each_field<F>([&](auto f) { csv_value(v.*(decltype(f)::mem_ptr), out, first); });
267 } else {
268 if (!first) out += ',';
269 first = false;
270 json_value(v, out); // scalar/array rendering is CSV-safe (no commas)
271 }
