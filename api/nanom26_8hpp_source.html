2// nanom26 — C++26 P2996 static-reflection provider for nanom's describe<T> seam.
3//
4// This is the macro-free nanom. The core library (nanom.hpp) consumes struct metadata through one
5// seam: `describe<T>::fields()` -> std::tuple<detail::fld<Name, MemPtr>...> and
6// `describe<T>::name()`. Under C++23 that tuple is synthesized by the NANOM_DESCRIBE preprocessor
7// macro (describe_macro.hpp); here the SAME tuple is synthesized from the type itself with P2996
8// reflection, so any eligible struct is usable with strct<>, overlay<>, soa<>, schema_of<>,
9// to_json, csv, bulk_decode — with no registration line at all:
10//
11// struct eth_hdr { std::array<std::uint8_t,6> dst, src; nanom::be<std::uint16_t> eth_type; };
12// // ...that's it. No NANOM_DESCRIBE. It is already Described.
13//
14// Eligibility (nanom::Reflectable): a named, non-union class type with no base classes and at
15// least one non-static data member, where every member is public, named, and not a C++ bit-field
16// (nanom's ubits<>/ibits<> wrappers are ordinary members and are fine) — and whose namespace
17// chain does not root in `std` or `nanom`. The last rule keeps library internals (std::array,
18// nanom::be<>, nanom::ubits<>) out of the Described concept so the core's if-constexpr dispatch
19// (array columns, wire scalars) keeps its C++23 meaning.
20//
21// Escape hatches: an explicit NANOM_DESCRIBE_FORCE_MACRO specialization always beats this partial
22// specialization (use it to register a member subset); -DNANOM_HAS_REFLECTION=0 disables this
23// provider entirely.
24//
25// Fork-facing surface (kept deliberately tiny; the P2996 API still moves between fork snapshots):
26// nonstatic_data_members_of, access_context::unchecked, identifier_of, has_identifier, parent_of,
27// bases_of, is_class_type/is_union_type/is_namespace/is_bit_field/is_public, and the ^^ / [: :]
28// operators. Everything else is plain C++.

69consteval bool rooted_in_library_namespace(std::meta::info r) {
70 std::meta::info cur = r;
71 std::meta::info up = std::meta::parent_of(cur);
72 while (up != ^^::) {
73 cur = up;
74 up = std::meta::parent_of(cur);
75 }
76 if (!std::meta::is_namespace(cur) || !std::meta::has_identifier(cur)) return false;
77 const std::string_view id = std::meta::identifier_of(cur);
78 return id == "std" || id == "nanom";
79}

82template <class T>
83consteval bool reflectable() {
84 const std::meta::info t = ^^T;
85 if (!std::meta::is_class_type(t) || std::meta::is_union_type(t)) return false;
86 if (!std::meta::has_identifier(t)) return false; // lambdas, anonymous structs, specializations
87 if (rooted_in_library_namespace(t)) return false;
88 if (!std::meta::bases_of(t, ctx()).empty()) return false;
89 const auto members = std::meta::nonstatic_data_members_of(t, ctx());
90 if (members.empty()) return false;
91 for (const std::meta::info m : members) {
92 if (!std::meta::has_identifier(m)) return false;
93 if (std::meta::is_bit_field(m)) return false; // no member pointer exists for these
94 if (!std::meta::is_public(m)) return false;
95 }
96 return true;
97}

120template <class T>
121struct name_holder {
122 private:
123 static consteval std::size_t qualified_len() {
124 std::size_t n = std::meta::identifier_of(^^T).size();
125 for (auto p = std::meta::parent_of(^^T); p != ^^::; p = std::meta::parent_of(p)) {
126 if (std::meta::has_identifier(p)) n += std::meta::identifier_of(p).size() + 2;
127 }
128 return n;
129 }
130 template <std::size_t N>
131 static consteval fixed_string<N + 1> qualified_name() {
132 fixed_string<N + 1> out{};
133 std::size_t pos = N; // fill right-to-left: we walk inner -> outer
134 const auto put = [&](std::string_view s) {
135 pos -= s.size();
136 for (std::size_t i = 0; i < s.size(); ++i) out.data[pos + i] = s[i];
137 };
138 put(std::meta::identifier_of(^^T));
139 for (auto p = std::meta::parent_of(^^T); p != ^^::; p = std::meta::parent_of(p)) {
140 if (std::meta::has_identifier(p)) {
141 put("::");
142 put(std::meta::identifier_of(p));
143 }
144 }
145 return out;
146 }
147
148 public:
149 static constexpr auto value = qualified_name<qualified_len()>();
150};
