1// ============================================================================
2// nanom — a nom-inspired header-only C++23 binary parser combinator library
3// with struct registration, schema generation and columnar (SoA) storage.
4//
5// https://github.com/yoavbendor/nanom
6//
7// This is the UMBRELLA header: it includes the whole library. The code lives in
8// layered headers you can also include individually (see README "Library layout"):
9//
10// prelude.hpp std includes, NANOM_HD, feature probes (shared config)
11// nom.hpp the pure rust-nom parallel: input/result, combinators,
12// binary + text numbers, bits (parser-only subset)
13// reflect.hpp fixed_string, wire types, describe<T> seam, strct<>/overlay<>
14// schema.hpp schemas + Arrow/Avro/JSON/CSV emission (extra)
15// soa.hpp columnar Struct-of-Arrays storage (extra)
16// bulk.hpp data-parallel (GPU-ready) scatter (separate, opt-in)
17//
18// Quick start (see README.md for more copy-paste examples):
19//
20// #include <nanom/nanom.hpp>
21// namespace nm = nanom;
22//
23// struct eth_hdr {
24// std::array<std::uint8_t,6> dst, src;
25// nm::be<std::uint16_t> eth_type;
26// };
27// NANOM_DESCRIBE(eth_hdr, dst, src, eth_type); // (optional under C++26 reflection)
28//
29// nm::input in = nm::from(buffer);
30// auto r = nm::strct<eth_hdr>()(in); // parse by value
31// if (!r) { std::puts(r.error().render(in).c_str()); return; }
32// std::uint16_t t = r->value.eth_type; // decoded, host order
33// nm::input rest = r->rest; // zero-copy remainder
34//
35// Everything a rust-nom user knows exists under the same name (see
36// docs/CHEATSHEET.md): tag, take, take_while1, alt, opt, many0, preceded,
37// delimited, separated_list0, length_data, map, verify, cut, context, be_u16…
38//
39// SPDX-License-Identifier: Apache-2.0
40// ============================================================================
